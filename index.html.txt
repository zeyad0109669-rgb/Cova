    <script>
        let products = JSON.parse(localStorage.getItem('covaProducts')) || [];
        let selectedSize = '';

        // تحديث عرض المنتجات
        function renderProducts() {
            const grid = document.getElementById('productsGrid');
            grid.innerHTML = '';

            products.forEach((product, index) => {
                const productCard = document.createElement('div');
                productCard.className = 'product-card';
                productCard.innerHTML = `
                    <img src="${product.image}" alt="${product.name}" class="product-image" onclick="openDetail(${index})">
                    <div class="product-info">
                        <h3 class="product-title">${product.name}</h3>
                        <div class="product-price">
                            ${product.price} جنيه
                            ${product.oldPrice ? `<span class="old-price">${product.oldPrice} جنيه</span>` : ''}
                        </div>
                        <div class="sizes">
                            ${product.sizes.map(size => 
                                `<span class="size-btn" onclick="selectSize('${size}', this)">${size}</span>`
                            ).join('')}
                        </div>
                        <button class="whatsapp-btn" onclick="orderProduct(${index})">
                            <i class="fab fa-whatsapp"></i> اطلب الآن
                        </button>
                    </div>
                `;
                grid.appendChild(productCard);
            });
        }

        // فتح لوحة الإدارة
        function toggleAdmin() {
            const panel = document.getElementById('adminPanel');
            panel.classList.toggle('active');
        }

        // إضافة منتج جديد
        document.getElementById('productForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const newProduct = {
                name: document.getElementById('productName').value,
                price: parseInt(document.getElementById('productPrice').value),
                oldPrice: document.getElementById('oldPrice').value ? parseInt(document.getElementById('oldPrice').value) : null,
                sizes: document.getElementById('sizes').value.split(',').map(s => s.trim()),
                image: document.getElementById('productImage').value
            };

            products.push(newProduct);
            localStorage.setItem('covaProducts', JSON.stringify(products));
            renderProducts();

            // إعادة تعيين النموذج
            this.reset();
            alert('✅ تم إضافة المنتج بنجاح!');
        });

        // اختيار مقاس
        function selectSize(size, button) {
            document.querySelectorAll('.size-btn').forEach(btn => btn.classList.remove('active'));
            button.classList.add('active');
            selectedSize = size;
        }

        // فتح تفاصيل المنتج
        function openDetail(index) {
            const product = products[index];
            document.getElementById('detailImage').src = product.image;
            document.getElementById('detailTitle').textContent = product.name;
            document.getElementById('detailPrice').innerHTML = `
                ${product.price} جنيه
                ${product.oldPrice ? `<span style="text-decoration: line-through; color: #888; font-size: 1.5rem; margin-right: 10px;">${product.oldPrice} جنيه</span>` : ''}
            `;
            
            const sizesContainer = document.getElementById('detailSizes');
            sizesContainer.innerHTML = product.sizes.map(size => 
                `<span class="size-btn" style="margin: 5px; padding: 12px 20px; font-size: 1.1rem;" 
                   onclick="selectDetailSize('${size}', this)">${size}</span>`
            ).join('');
            
            selectedSize = '';
            document.getElementById('productDetail').classList.add('active');
        }

        // اختيار مقاس في صفحة التفاصيل
        function selectDetailSize(size, button) {
            document.querySelectorAll('#detailSizes .size-btn').forEach(btn => btn.classList.remove('active'));
            button.classList.add('active');
            selectedSize = size;
        }

        // إغلاق تفاصيل المنتج
        function closeDetail() {
            document.getElementById('productDetail').classList.remove('active');
        }

        // طلب المنتج عبر واتساب
        function orderProduct(index) {
            if (!selectedSize) {
                alert('⚠️ يرجى اختيار المقاس أولاً');
                return;
            }
            
            const product = products[index];
            const message = `مرحباً 👋%0A%0Aأريد طلب:%0A${product.name}%0Aالمقاس: ${selectedSize}%0Aالسعر: ${product.price} جنيه%0A%0Aشكراً COVA ✨`;
            
            const whatsappUrl = `https://wa.me/201096694034?text=${message}`;
            window.open(whatsappUrl, '_blank');
        }

        // تحميل المنتجات عند بدء الصفحة
        renderProducts();

        // إغلاق التفاصيل عند الضغط خارجها
        document.getElementById('productDetail').addEventListener('click', function(e) {
            if (e.target === this) closeDetail();
        });
    </script>
